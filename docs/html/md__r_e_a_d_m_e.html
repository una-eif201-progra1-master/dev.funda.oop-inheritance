// Example of Inheritance
class Vehicle {
public:
string brand = "Ford";
void honk() {
cout << "Tuut, tuut!" << endl;
}
};
class Car: public Vehicle {
public:
string model = "Mustang";
};
int main() {
Car car1;
car1.honk(); // Output: Tuut, tuut!
cout << car1.brand + " " + car1.model; // Output: Ford Mustang
return 0;
}
